```python 
def level(number):
    level = number/10 + 1

    if number % 10 == 10
        level -=1
    if level > 21:
        level = 21
    
    return level

# 边界条件的优化处理
def level2(number):
    if number % 10 == 0:
        return number // 10
    else :
        return number//10 + 1
```
